# 串
* 串是数据元素类型必须是字符的线性表
* 比如 ： ""、"hello"、"你好"、"utf8"
* 串中任意连续的字符组成的串称为字串 
* 逻辑结构还是前驱后继这样的
* 存储结构分为链式存储（链串）和顺序存储（顺序串）

## 此处实现的是顺序串
## g++编译器
## Windows10 64位系统

# 数组
* 数组属于线性表
* 数组当中的每一个元素又是一个线性表

# 广义表
* 广义表是线性表的一种拓展
* 它的每一个元素可以是原子类型，也可以是广义表
* 这里所谓的原子类型是表示不可再分的，比如整数4就是原子类型，而线性表（5， 51， 20）可以在分成5， 51， 20这三个数，不是原子类型

# 以下的广义表的示例，来使得其更容易理解（来自[百度](https://baike.baidu.com/item/%E5%B9%BF%E4%B9%89%E8%A1%A8/3685109)）
① E=()  

E是一个空表，其长度为0。  

② L=(a，b)  

L是长度为2的广义表，它的两个元素都是原子，因此它是一个线性表  

③ A=(x，L)=(x，(a，b))  

A是长度为2的广义表，第一个元素是原子x，第二个元素是子表L。  

④ B=(A，y)=((x，(a，b))，y)  

B是长度为2的广义表，第一个元素是子表A，第二个元素是原子y。  

⑤ C=(A，B)=((x，(a，b))，((x，(a，b))，y))  

C的长度为2，两个元素都是子表。  

⑥ D=(a，D)=(a，(a，(a，(…))))  

D的长度为2，第一个元素是原子，第二个元素是D自身，展开后它是一个无限的广义表。

// 追加字符
void StrType::append(const char _str){
    // 满了就自动扩容一个
    if (length == maxsize){
        char *new_str = new char[maxsize+1];
        for (int i=0;i<length;i++){
            new_str[i] = str[i];
        }
        delete[] str;
        str = new_str;
        maxsize++;
    } else{}
    str[length] = _str;
    length++;
};

// 追加字符串
void StrType::append(const char *_str){
    // 没想另起方法，直接调用
    // 扩容添加字符串的大小
    int len = 0;
    while (_str[len] != '\0'){
        len++;
    }
    char *new_str = new char[maxsize+len];
    for (int i=0;i<length;i++){
        new_str[i] = str[i];
    }
    for (int j=0;j<len;j++){
        new_str[j+length] = _str[j];
    }
    length += len;
    maxsize += len;
    delete[] str;
    str = new_str;
};

// 插入字符
bool StrType::insert(const char _str, int location){
    // 先判断范围是否溢出
    if (location < 0 || location > length-1){
        return false;
    } else{
        // 扩容1个字符大小
        char *new_str = new char[maxsize];
        char *new_str_2 = new char[maxsize+1];
        for (int i=0;i<length;i++){
            new_str[i] = str[i];
        }

        // 另一个字符串数组用于重新赋值
        for (int i=0;i<length+1;i++){
            if (i < location){
                new_str_2[i] = str[i];
            } else if (i == location){
                new_str_2[location] = _str;
            } else{
                new_str_2[i] = str[i-1];
            }
        }
        delete[] str;
        str = new_str_2;
        maxsize++;
        length++;
        return true;
    }
};

bool StrType::remove(int location){
    // whether in range or len == 0
    if (location < 0 || location > length-1 || length == 0){
        return false;
    }
    else
    {
        char * another_str = new char[maxsize];
        for (int i=0;i<length-1;i++){
            if (i < location){
                another_str[i] = str[i];
            }else {
                another_str[i] = str[i+1];
            }
        }
        length--;
        delete[] str;
        str = another_str;
        return true;
    }
};

// 子串是否存在,又称为模式匹配(有BF算法，穷举思路,又称暴力破解算法、还有一个是KMP算法,这里用到是BF算法)
bool StrType::isExistBP(const char *_str) const {
    // 思路 ： 子串str1在主串str中同等长度的子串中匹配
    int child_len = 0;
    while (_str[child_len] != '\0'){
        child_len++ ;
    }
    // 先思考一个问题 ： 连续n个元素组成的有序数组中，取连续K个元素，有多少种？
    // 当然了，k<= n，答案是n-k+1
    for (int i=0;i<length-child_len+1;i++){
        // 每种情况逐个对比
        int same = 0;
        for (int j=0;j<child_len;j++){
            if (str[j+i] == _str[j]){
                same++;
            } else{
                break;
            }
            if (same == child_len){
                return true;
            } else{
                continue;
            }
        }
    }
    // 到这一步说明还是没找到
    return false;
};

// KMP算法
bool StrType::isExistKMP(const char *_str) const {
    // 获取子串长度
    int child_len = 0;
    while (_str[child_len] != '\0'){
        child_len++;
    }

    // next数组在KMP算法中被用于记录 要找的子串（又称为模式串）中各个前缀的最长公共前缀。
    // 它可以帮助在模式串和文本串匹配时跳过不必要的比较，提高匹配的效率。
    // get next
    int *next = new int[child_len];

    next[0] = 0;
    for (int i = 1, j = 0; i < child_len; i++ ) {
        while (j > 0 && _str[j] != _str[i]) {
            j = next[j - 1] ;
        }
        if (_str[j] == _str[i]) {
            j++;
        }
        next[i] = j;
    }

    int j = 0;
    for (int i = 0; i < length; i ++ ) {
        while (j > 0 && _str[j] != str[i]) j = next[j - 1];
        if (_str[j] == str[i]) {
            j++;
        }
        if (j == child_len) {
            // 找到子串
            return true;
        }
    }
    // 到这里表明不存在
    return false;
}

// 通过ASCII找出小写字母(97~122)
char * StrType::upper(){
    char *__str = new char[length];
    for (int i=0;i<length;i++){
        int to_ascii = (int)str[i];
        if (to_ascii >= 97 && to_ascii <= 122){
            __str[i] = (char)(to_ascii-122+90);
        } else{
            __str[i ] = str[i];
        }
    }return __str;
}

// 内置英文字母全小写
// 大写(65~90)
char * StrType::lower(){
    char *__str = new char[length];
    for (int i=0;i<length;i++){
        int to_ascii = (int)str[i];
        if (to_ascii >= 65 && to_ascii <= 90){
            __str[i] = (char)(to_ascii+122-90);
        } else{
            __str[i ] = str[i];
        }
    }return __str;
}

// 重写+号操作，使得串能够进行相加运算，分为下面两种,用的是C++的泛型编程特性
// 重载 + 运算符，用于把两个 Str实例 相加
StrType StrType::operator+(const StrType &anoter_strType) {
    int new_maxsize = anoter_strType.maxsize + maxsize;
    int new_length = anoter_strType.length + length;
    char *_str = new char [new_maxsize];
    for (int i=0;i<length;i++){
        _str[i] = str[i];
    }
    for (int j=0;j<anoter_strType.length;j++){
        _str[j+length] = anoter_strType.str[j];
    }
    StrType new_str;
    new_str.length = new_length;
    new_str.maxsize - new_maxsize;
    new_str.str = _str;
    return new_str;
}

// 拷贝函数
StrType::StrType(const StrType &exa) {
    length = exa.length;
    maxsize = exa.maxsize;
    str = exa.str;
};
