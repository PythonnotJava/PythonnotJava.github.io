    transpose

    Matrix transpose()

    获取矩阵对应的转置矩阵

    test

    import 'package:flutter_matrix/matrix_type.dart';

main() {
  data_format = "%2.0f";
  var mt = Matrix.fromList([
    [1, 4, 5],
    [3, -2, 1],
    [9, 0, 1]
  ]);
  mt.transpose().visible();
}

    output

    [
 [  1   3   9]
 [  4  -2   0]
 [  5   1   1]
]

    trace

    获取矩阵最小维度下主对角线上数据的和，方阵中的迹

    import 'package:flutter_matrix/matrix_type.dart';

main() {
  data_format = "%2.0f";
  var mt = Matrix.fromList([
    [1, 4, 5],
    [3, -2, 1],
    [9, 0, 1],

    0.0

    coincidental

    Matrix coincidental({required int row, required int column})

    获取余子式，即去掉某行和某列而拼成的矩阵

    test

    import 'package:flutter_matrix/matrix_type.dart';

main() {
  data_format = "%2.0f";
  var mt = Matrix.fromList([
    [1, 4, 5],
    [3, -2, 1],
    [9, 0, 1],
    [5, 3, 2]
  ]);
  mt.coincidental(row: 0, column: 1).visible();
}

    output

    [
 [  3   1]
 [  9   1]
 [  5   2]
]

    elementary_exchange

    void elementary_exchange({required int index1, required int index2, bool horizontal = true})

    初等变换——交换行或者列

    test

    import 'package:flutter_matrix/matrix_type.dart';

main() {
  data_format = "%2.0f";
  var mt = Matrix.fromList([
    [1, 4, 5],
    [3, -2, 1],
    [9, 0, 1],
    [5, 3, 2]
  ]);
  mt
    ..elementary_exchange(index1: 3, index2: 2, horizontal: true)
    ..visible()
    ..elementary_exchange(index1: 0, index2: 1, horizontal: false)
    ..visible();
}

    output

    [
 [  1   4   5]
 [  3  -2   1]
 [  5   3   2]
 [  9   0   1]
]
[
 [  4   1   5]
 [ -2   3   1]
 [  3   5   2]
 [  0   9   1]
]

    elementary_add

    void elementary_add({required int index1, required int index2, required double number, bool horizontal = true})

    dot

    Matrix dot({required Matrix other})

    左行右列做矩阵内积，必须满足$A_(mn)$和$B_(nk)$的形状

    test

    import 'package:flutter_matrix/matrix_type.dart';

main() {
  data_format = "%2.2f";
  var mt = Matrix.fromList([
    [1, 4, 5],
    [3, -2, 1],
    [9, 0, 1],
    [5, 3, 2]
  ]);
  var mt1 = Matrix.fromList(
    [[1, 4, 5],
    [3, -2, 1],
    [9, 0, 1],]
  );
  mt.dot(other: mt1).visible();
}

    output

    [
 [58.00 -4.00 14.00]
 [ 6.00 16.00 14.00]
 [18.00 36.00 46.00]
 [32.00 14.00 30.00]
]

    double get det

main() {
  data_format = "%3.2f";
  var mt = Matrix.fromList([
    [1, 4, 5, 5],
    [3, -2, 1, 3],

    int get rank

    矩阵的秩

    test

    import 'package:flutter_matrix/matrix_type.dart';

main() {
  data_format = "%5.2f";
  var mt = Matrix.fromList([
    [1, 4, 5, 5],
    [3, -2, 1, 3],
    [9, 0, 1, 2],
    [5, 3, 2, -5]
  ]);
  print(mt.rank);
  mt[3] = [1 - 9, 4 - 0, 5 - 1, 5 - 2]; // row_index0 - row_index2
  print(mt.det);
  print(mt.rank);
}

    output

    4
0.0
